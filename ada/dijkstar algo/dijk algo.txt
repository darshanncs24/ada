#include <stdio.h>

#define V 4
#define INF 999

int mini(int d[], int visited[]) {
    int min = INF, index = -1;
    int i;

    for(i = 0; i < V; i++) {
        if(visited[i] == 0 && d[i] <= min) {
            min = d[i];
            index = i;
        }
    }

    return index;
}

void dijkstra(int g[V][V], int src) {
    int d[V];
    int visited[V];
    int i, j, u;

    for(i = 0; i < V; i++) {
        d[i] = INF;
        visited[i] = 0;
    }

    d[src] = 0;

    for(i = 0; i < V - 1; i++) {

        u = mini(d, visited);
        visited[u] = 1;

        for(j = 0; j < V; j++) {
            if(!visited[j] && g[u][j] &&
               d[u] != INF &&
               d[u] + g[u][j] < d[j]) {

                d[j] = d[u] + g[u][j];
            }
        }
    }

    printf("Vertex \t Distance from Source\n");
    for(i = 0; i < V; i++) {
        printf("%d \t %d\n", i, d[i]);
    }
}

int main() {

    int g[V][V] = {
        {0, 1, 4, 0},
        {1, 0, 2, 6},
        {4, 2, 0, 3},
        {0, 6, 3, 0}
    };

    int src;

    printf("Enter source vertex (0-3): ");
    scanf("%d", &src);

    dijkstra(g, src);

    return 0;
}
