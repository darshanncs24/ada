#include <stdio.h>

#define MAX 20
#define INF 999

void floyds(int a[MAX][MAX], int n) {
    int i, j, k;

    for(k = 0; k < n; k++) {
        for(i = 0; i < n; i++) {
            for(j = 0; j < n; j++) {

                if(a[i][j] > a[i][k] + a[k][j])
                    a[i][j] = a[i][k] + a[k][j];
            }
        }
    }
}

int main() {
    int a[MAX][MAX];
    int n, i, j;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");

    for(i = 0; i < n; i++) {
        for(j = 0; j < n; j++) {
            scanf("%d", &a[i][j]);

            if(i != j && a[i][j] == 0)
                a[i][j] = INF;
        }
    }

    floyds(a, n);

    printf("All Pair Shortest Path Matrix:\n");

    for(i = 0; i < n; i++) {
        for(j = 0; j < n; j++) {

            if(a[i][j] == INF)
                printf("INF ");
            else
                printf("%d ", a[i][j]);
        }

        printf("\n");
    }

    return 0;
}
