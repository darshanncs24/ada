#include <stdio.h>

#define MAX 20

int knapsack(int n, int w[], int p[], int m) {
    int table[MAX][MAX];
    int i, j;

    for(i = 0; i <= n; i++) {
        for(j = 0; j <= m; j++) {

            if(i == 0 || j == 0)
                table[i][j] = 0;

            else if(j < w[i - 1])
                table[i][j] = table[i - 1][j];

            else {
                int include = p[i - 1] + table[i - 1][j - w[i - 1]];
                int exclude = table[i - 1][j];

                if(include > exclude)
                    table[i][j] = include;
                else
                    table[i][j] = exclude;
            }
        }
    }

    return table[n][m];
}

int main() {
    int n, m, i;
    int w[MAX], p[MAX];
    int profit;

    printf("Enter number of items: ");
    scanf("%d", &n);

    printf("Enter weights:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &w[i]);
    }

    printf("Enter profits:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &p[i]);
    }

    printf("Enter knapsack capacity: ");
    scanf("%d", &m);

    profit = knapsack(n, w, p, m);

    printf("Maximum profit = %d\n", profit);

    return 0;
}
