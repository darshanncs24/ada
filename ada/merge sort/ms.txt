#include <stdio.h>
#include <time.h>

#define MAX 100

void merge(int a[], int low, int mid, int high) {
    int i, j, k;
    int temp[MAX];

    i = low;
    j = mid + 1;
    k = low;

    while(i <= mid && j <= high) {
        if(a[i] < a[j])
            temp[k++] = a[i++];
        else
            temp[k++] = a[j++];
    }

    while(i <= mid)
        temp[k++] = a[i++];

    while(j <= high)
        temp[k++] = a[j++];

    for(i = low; i <= high; i++)
        a[i] = temp[i];
}

void ms(int a[], int low, int high) {
    int mid;

    if(low < high) {
        mid = (low + high) / 2;

        ms(a, low, mid);
        ms(a, mid + 1, high);

        merge(a, low, mid, high);
    }
}

int main() {
    int a[MAX], n, i;
    clock_t start, end;
    double time_taken;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter elements:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    start = clock();

    ms(a, 0, n - 1);

    end = clock();

    time_taken = ((double)(end - start))  

    printf("Sorted elements are:\n");
    for(i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }

    printf("\nTime taken = %f seconds\n", time_taken);

    return 0;
}
