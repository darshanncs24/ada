#include <stdio.h>
#include <time.h>

#define MAX 100

int part(int a[], int low, int high) {
    int pivot, i, j, temp;

    pivot = a[low];
    i = low + 1;
    j = high;

    while(i <= j) {

        while(i <= high && a[i] <= pivot)
            i++;

        while(a[j] > pivot)
            j--;

        if(i < j) {
            temp = a[i];
            a[i] = a[j];
            a[j] = temp;
        }
    }

    temp = a[low];
    a[low] = a[j];
    a[j] = temp;

    return j;
}

void quick(int a[], int low, int high) {
    int p;

    if(low < high) {
        p = part(a, low, high);

        quick(a, low, p - 1);
        quick(a, p + 1, high);
    }
}

int main() {
    int a[MAX], n, i;
    clock_t start, end;
    double time_taken;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter elements:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    start = clock();

    quick(a, 0, n - 1);

    end = clock();

    time_taken = ((double)(end - start))  
    printf("Sorted elements are:\n");
    for(i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }

    printf("\nTime taken = %f seconds\n", time_taken);

    return 0;
}
